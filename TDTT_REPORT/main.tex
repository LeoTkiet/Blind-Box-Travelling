\section{Phân tích \& Xác định bài toán}
\subsection{Bối cảnh thực tế}
Giới trẻ hiện nay thường gặp tình trạng "quá tải sự lựa chọn" khi quyết định địa điểm ăn uống, giải trí (hội chứng FOMO và mệt mỏi khi quyết định). Xu hướng "Hộp mù" (Blind box) đang thịnh hành vì mang lại cảm giác bất ngờ và thú vị. Việc kết hợp bản đồ số, AI và trải nghiệm "hộp mù" giải quyết trực tiếp nhu cầu tìm kiếm sự mới mẻ một cách tiện lợi.

\subsection{Xác định Input / Output}
Bài toán được chuyển từ yêu cầu mơ hồ sang các tham số có thể xử lý rõ ràng:
\begin{itemize}
    \item \textbf{Input:} 
    \begin{itemize}
        \item Tọa độ GPS hiện tại của Client: $(Lat_{current}, Lng_{current})$.
        \item Bán kính di chuyển tối đa: $R$ (km).
        \item Thể loại yêu thích: $Category$ (VD: Cafe, Ăn vặt).
        \item Dữ liệu nhóm (Group mode): Danh sách sở thích của $n$ thành viên.
    \end{itemize}
    \item \textbf{Output:} 
    \begin{itemize}
        \item Tọa độ điểm đến duy nhất (bị ẩn tên): $(Lat_{target}, Lng_{target})$.
        \item Đoạn văn 30 từ do AI gợi ý về không gian (gợi sự tò mò).
        \item Danh sách vật dụng cần mang (dựa trên thời tiết thực tế).
        \item Quỹ tiền dự kiến phải chia đều cho nhóm.
    \end{itemize}
\end{itemize}

\section{Phân rã (Decomposition) \& Thiết kế Module}
Hệ thống được thiết kế theo kiến trúc Microservices/Serverless và được phân rã thành các module logic chạy độc lập nhưng liên kết chặt chẽ với nhau. Dựa trên nghiệp vụ, hệ thống bao gồm 8 Module chính:
\begin{enumerate}
    \item \textbf{Thu thập (Module 1):} Cào dữ liệu offline bằng Python $\rightarrow$ Làm sạch và chuẩn hóa $\rightarrow$ Tải lên Supabase (Data Warehouse).
    \item \textbf{Input (Module 2 \& 6):} Sử dụng Geolocation.getCurrentPosition API lấy tọa độ; Ứng dụng Supabase Realtime để gom nhóm người dùng thành viên và tính trung bình sở thích.
    \item \textbf{Lọc \& Xếp hạng (Module 3):} Hoạt động dựa trên thuật toán "Hidden Gem", query trực tiếp vào Database, lọc theo bán kính yêu cầu và tính điểm xếp hạng để tìm ra điểm duy nhất.
    \item \textbf{Gia tăng giá trị trải nghiệm (Module 4 \& 5):} Hệ thống kết hợp thông tin dự báo thời tiết từ OpenWeatherMap cùng với Gemini API (LLM) để sinh ra Content mồi, tạo sự tò mò.
    \item \textbf{Hiển thị \& Điều hướng (Module 7):} Tích hợp thư viện Mapbox API để dẫn đường trên bản đồ tương tác, dùng Canvas-confetti tạo hiệu ứng chúc mừng (pháo hoa) khi khoảng cách $< 200m$.
    \item \textbf{Tìm kiếm thông minh - Smart Search (Module 8):} Đánh giá và lọc các category cũng như keywords dựa trên Tagging thông minh từ AI, mang lại khả năng phân loại và truy vấn dữ liệu linh hoạt.
\end{enumerate}

\section{Các Hàm và Thuật toán trong Hệ Thống}
\subsection{Các Thuật toán tại Module 1 \& Module 8 (Data Pipeline \& Smart Search)}
\begin{itemize}
    \item \textbf{Exponential Backoff \& Jitter:} Thuật toán quản lý tải khi gặp lỗi từ các API nhà cung cấp. Thời gian chờ tăng theo hàm mũ và cộng thêm yếu tố ngẫu nhiên nhằm tránh tình trạng gửi request đồng loạt khi server API mở lại. Dùng trong file \texttt{enricher.js}.
    \item \textbf{Token Bucket (Rate Limiter):} Giới hạn số lượng request theo đơn vị thời gian cho các API Client để không vi phạm chính sách của Google (Gemini API)/Groq.
    \item \textbf{JSON Parse/Repair:} Xử lý sâu vào Response của AI. Dùng cấu trúc vòng lặp Match để tìm đối tượng JSON hợp lệ bị đưa lẫn vào markdown; nếu hỏng sẽ quay lui ép AI sửa format.
\end{itemize}

\subsection{Luồng Random \& Gợi ý (Hidden Gem)}
\subsubsection{Hàm tính khoảng cách Haversine}
Sử dụng phương pháp Haversine để tính khoảng cách đường chim bay giữa hai điểm trên bề mặt hình cầu dựa vào vĩ độ và kinh độ.
\begin{equation}
a = \sin^2\left(\frac{\Delta Lat}{2}\right) + \cos(Lat_1) \cdot \cos(Lat_2) \cdot \sin^2\left(\frac{\Delta Lng}{2}\right)
\end{equation}
\begin{equation}
d = R \cdot 2 \cdot \text{atan2}(\sqrt{a}, \sqrt{1-a})
\end{equation}
Trong đó $R = 6371$ km là bán kính trung bình của Trái Đất.

\subsubsection{Thuật toán tính điểm Hộp mù (Scoring Algorithm)}
Thuật toán được thiết kế để đưa ra điểm $TotalScore$ cân bằng giữa chất lượng (Quality), độ bí ẩn (Anonymity) và tính bất ngờ (Random).
\begin{itemize}
    \item \textbf{Quality Score ($S_q$) theo Bayesian Average:} Nhằm khắc phục trường hợp các địa điểm có ít đánh giá nhưng đạt 5 sao cao đột biến. Điểm số được tinh chỉnh kéo về hệ số trung bình của khu vực ($m$) kết hợp tham số tự tin ($C$).
    \begin{equation}
    S_q = \left[ \frac{\text{reviews\_count} \times \text{rating} + C \times m}{\text{reviews\_count} + C} \right] \div 5.0
    \end{equation}
    \item \textbf{Anonymity Score ($S_a$):} Điểm địa điểm phải đáp ứng lượng review cận dưới (ngưỡng $\ge 10$) để đảm bảo điểm đến hiện hữu. Tuy nhiên, nếu có quá ít lượt đánh giá so với max reviews vùng thì điểm ẩn dật sẽ càng cao để kích thích khám phá.
    \begin{equation}
    S_a = 1 - \frac{\log_{10}(\text{reviews\_count} + 1)}{\log_{10}(\text{max\_reviews\_in\_area} + 1)}
    \end{equation}
    \item \textbf{Random Score ($S_r$):} Điểm ngẫu nhiên tạo độ nhiễu an toàn. Dùng \texttt{Math.random()} cấu thành.
\end{itemize}
\textbf{Công thức tổng hợp Scoring:}
\begin{equation}
\text{TotalScore} = (S_q \times 0.45) + (S_a \times 0.30) + (S_r \times 0.25)
\end{equation}
Điểm có Total Score cao nhất trong kết quả filter sẽ được trả về cho người dùng.

\section{Các Công Cụ Sử Dụng Trong Module}
\begin{itemize}
    \item \textbf{Python \& Selenium WebDriver:} Công cụ tự động hóa trong module Crawler để giả lập thao tác của người dùng lướt bản đồ số và thu thập dữ liệu DOM động từ client.
    \item \textbf{Node.js \& Các SDK (\texttt{@google/generative-ai}, \texttt{groq-sdk}):} Sử dụng trong module Enricher để đẩy dữ liệu phân loại bất đồng bộ siêu tốc.
    \item \textbf{Các mô hình ngôn ngữ lớn (LLMs):} \texttt{gemini-2.5-flash} (Google), \texttt{llama-3.3-70b-versatile} (Groq), và các public models qua \textbf{OpenRouter}. Chúng dùng để cào nhãn thực thể (Analyst) và review kiểm soát nội dung (Supervisor).
    \item \textbf{Next.js (React Framework):} Framework gốc xử lý Frontend UX/UI, Routing và Backend Serverless cho các truy xuất API.
    \item \textbf{Supabase (PostgreSQL Database):} Nơi lưu trữ đám mây cung cấp các truy vấn bảng phức tạp và Row Level Security bảo mật dữ liệu.
    \item \textbf{Phụ trợ API:} \textbf{Mapbox} (hiển thị, marker bản đồ động) và \textbf{OpenWeatherMap API} (bản tin thời tiết thời gian thực phục vụ content AI của dự án).
    \item \textbf{TailwindCSS \& Framer Motion:} Xây dựng hệ thống UI tối giản với hiệu ứng mượt mà.
\end{itemize}

\section{Tiến trình Thực hiện (Theo phân rã Module)}
Dựa trên kiến trúc 8 module đã đề ra, tiến độ thực tế được ghi nhận đến hiện tại như sau:
\begin{itemize}
    \item \textbf{Module 1 (Thu thập - Hoàn thành 100\%):} Đã viết xong mã nguồn \texttt{crawl\_data.py} tự động lấy thông tin từ Google Maps bằng Selenium. Làm giàu, gán nhãn tags thành công \texttt{data\_enriched.json} với AI (\texttt{enricher.js}) và đẩy đủ dữ liệu mẫu lên kho chứa hệ thống Supabase.
    \item \textbf{Module 2 \& 6 (Input \& Group Mode - Đang phát triển):} Đã lấy thành công \texttt{lat, lng} của người dùng qua HTML5 Geolocation (Module 2). Về tính năng Gom nhóm và tính trung bình sở thích qua Supabase Realtime (Module 6), hiện đang hoàn tất thiết kế kiến trúc và chuẩn bị code realtime.
    \item \textbf{Module 3 (Lọc \& Xếp hạng - Hoàn thành 100\%):} Hoàn thiện hàm API \texttt{/api/blind-box}. Trích xuất thành công dữ liệu từ DB, loại bỏ địa điểm ngoài bán kính thực tế bằng công thức Haversine; tính điểm và trả về \texttt{TotalScore} chính xác nhờ áp dụng hệ số đánh giá tĩnh Bayesian Average kết hợp Logarit.
    \item \textbf{Module 4 \& 5 (Gia tăng giá trị trải nghiệm - Hoàn thành 100\%):} Đã thiết lập xong Engine Prompt tại \texttt{/api/ai-experience}. Fetch đồng thời dữ liệu OpenWeatherMap phục vụ logic "thời tiết hiện tại" và gọi LLM thông qua Groq API trả về các đoạn phân tích tò mò mập mờ hoàn hảo.
    \item \textbf{Module 7 (Hiển thị \& Điều hướng - Đang hoàn thiện):} Đã nhúng xong thư viện Mapbox trên trang UI Next.js, tạo Marker thành công lên bản đồ ứng dụng. Tính năng dẫn đường chi tiết và phát động Canvas-confetti (khi tới gần $< 200m$) đang được tối ưu kiểm thử.
    \item \textbf{Module 8 (Smart Search - Chờ ráp UI):} Thuật toán gắn nhãn tìm kiếm (Tagging/Categorizing) trong dữ liệu đã phủ đủ và chuẩn hóa. Cấu trúc Back-end hỗ trợ lọc thông minh đã có, phần UI box Smart Search trên ứng dụng Frontend đang được chắp nối để chạy hoàn chỉnh.
\end{itemize}
